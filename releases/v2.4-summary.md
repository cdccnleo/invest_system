def get_ainvest_llm_client():

def call_llm_for_event_match_ainvest(event_topic, holdings, timeout=30):
    """PIT #28 强化: 持仓产业链 hint 内嵌"""
    biz_hints = "\n".join(
        f"  {c}: {_HOLDING_BIZ_HINTS.get(c, '')}"
        for c in holdings if c in _HOLDING_BIZ_HINTS
    )
    prompt = f"""
你是 A 股投资顾问, 分析事件对持仓的影响:
事件: {event_topic}

持仓 45 个, 关注以下持仓产业链关系:
{biz_hints}

返回 JSON: {{"affected_codes": [...], "direction": "positive|negative|neutral", "reasoning": "..."}}
"""
    client = get_ainvest_llm_client()
    result = client.chat(prompt, system="你是专业 A 股投资顾问")
    return _parse_llm_json(result["content"])  # PIT #28: 3 级容错
